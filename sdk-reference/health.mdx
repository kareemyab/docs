---
title: Health Check
description: 'Checks the operational status of the Trust Engine API server.'
---

## Function Signature

```typescript
// This is not a direct SDK function but a fetch call to the health endpoint.
async function healthCheck(): Promise<any>;
```

## Usage

<CodeGroup>
```typescript SDK (Recommended)
// Note: Health check is typically used for monitoring, not end-user applications
// Most applications won't need to call this endpoint directly

import { fetch } from 'node-fetch'; // or use built-in fetch in modern environments

async function checkServerHealth() {
  try {
    const response = await fetch('https://core-api-server.onrender.com/health');
    const data = await response.json();
    
    console.log('Server status:', data.status);
    console.log('Timestamp:', data.timestamp);
    
    return data;
  } catch (error) {
    console.error('Health check failed:', error.message);
    throw error;
  }
}

// Usage
checkServerHealth()
  .then(result => console.log('Server is healthy:', result))
  .catch(error => console.error('Server health issue:', error));
```
</CodeGroup>
