# Type Definitions

This page provides detailed information about the type definitions used in the Trust Engine SDK.

## `createWallet`

### Options

<ResponseField name="userID" type="string" required>
  Unique identifier for the user.
</ResponseField>

### Response

<ResponseField name="message" type="string">
  A success message.
</ResponseField>

<ResponseField name="details" type="object">
  An object containing the details of the created wallet.
  <Expandable>
    <ResponseField name="walletAddress" type="string">
      The address of the new wallet.
    </ResponseField>
    <ResponseField name="chain" type="string">
      The blockchain the wallet was created on.
    </ResponseField>
    <ResponseField name="transactionSignature" type="string">
      The signature of the transaction that created the wallet.
    </ResponseField>
    <ResponseField name="explorerUrl" type="string">
      A URL to view the transaction on a block explorer.
    </ResponseField>
    <ResponseField name="userID" type="string">
      User ID connected to the wallet address
    </ResponseField>
  </Expandable>
</ResponseField>

## `registerFile`

### Options

<ResponseField name="file" type="File" required>
  The file to register.
</ResponseField>

<ResponseField name="contentTitle" type="string" required>
  The title of the content.
</ResponseField>

<ResponseField name="walletAddress" type="string" required>
  The address of the wallet to register the file with.
</ResponseField>

<ResponseField name="walletType" type="'self' | 'managed'" required>
  The type of wallet to use.
</ResponseField>

<ResponseField name="metadata" type="string" optional>
  Additional metadata to store with the file.
</ResponseField>

<ResponseField name="signTransaction" type="(transaction: Transaction) => Promise<Transaction>" optional>
  A function to sign a transaction. Usually a function from your wallet provider (Phantom, Metamask, etc. )
</ResponseField>

### Response

The response from registering a file. This can vary depending on whether the wallet type is `'self'` or `'managed'`.

#### For Managed Wallets (200 OK)

<ResponseField name="message" type="string">
  A success message.
</ResponseField>

<ResponseField name="details" type="object">
  An object containing the details of the registration.
  <Expandable>
    <ResponseField name="status" type="string">
      The status of the registration.
    </ResponseField>
    <ResponseField name="transactionSignature" type="string">
      The signature of the transaction that registered the file.
    </ResponseField>
    <ResponseField name="contentRegistrationPDA" type="string">
      The Program Derived Address for the registration.
    </ResponseField>
    <ResponseField name="contentHash" type="string">
      The SHA256 hash of the file content.
    </ResponseField>
    <ResponseField name="ipfsCid" type="string">
      The IPFS CID of the file.
    </ResponseField>
    <ResponseField name="explorerUrl" type="string">
      A URL to view the transaction on a block explorer.
    </ResponseField>
  </Expandable>
</ResponseField>

#### For Self Wallets (202 Accepted)

<ResponseField name="message" type="string">
  A success message.
</ResponseField>

<ResponseField name="details" type="object">
  An object containing the details of the registration.
  <Expandable>
    <ResponseField name="status" type="string">
      The status of the registration.
    </ResponseField>
    <ResponseField name="transaction" type="string">
      A base64 encoded transaction that needs to be signed by the client.
    </ResponseField>
    <ResponseField name="ipfsCid" type="string">
      The IPFS CID of the file.
    </ResponseField>
    <ResponseField name="contentRegistrationPDA" type="string">
      The Program Derived Address for the registration.
    </ResponseField>
    <ResponseField name="contentHash" type="string">
      The SHA256 hash of the file content.
    </ResponseField>
  </Expandable>
</ResponseField>

## `searchFile`

### Parameters

<ResponseField name="file" type="File" required>
  The file to search for.
</ResponseField>

<ResponseField name="walletAddress" type="string" optional>
  Optional wallet address to filter search results to specific registrations.
</ResponseField>

### Response

<ResponseField name="searchMode" type="string">
  The type of search performed.
</ResponseField>

<ResponseField name="message" type="string">
  A message indicating the search results.
</ResponseField>

<ResponseField name="searchCriteria" type="object">
  The search parameters that were used.
  <Expandable>
    <ResponseField name="contentHash" type="string">
      The content hash that was searched for.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="totalResults" type="number">
  Number of registration records found.
</ResponseField>

<ResponseField name="registrations" type="Array<object>">
  An array of objects, where each object contains the details of a registration record.
  <Expandable>
    <ResponseField name="title" type="string">
      The chosen title of the registration
    </ResponseField>
    <ResponseField name="contentHash" type="string">
      The SHA256 hash of the file content.
    </ResponseField>
    <ResponseField name="pdaAddress" type="string">
      The Program Derived Address for the registration.
    </ResponseField>
    <ResponseField name="ipfsCid" type="string">
      The IPFS Content Identifier for metadata storage.
    </ResponseField>
    <ResponseField name="registeredBy" type="string">
      The wallet address that registered the file.
    </ResponseField>
    <ResponseField name="userID" type="string | null">
      The user ID associated with the creator of the file (null if no relation found).
    </ResponseField>
    <ResponseField name="timestamp" type="string">
      ISO timestamp when file was registered.
    </ResponseField>
    <ResponseField name="status" type="string">
      Status of the registration.
    </ResponseField>
    <ResponseField name="explorerUrl" type="string">
      Link to view on Solana Explorer.
    </ResponseField>
    <ResponseField name="metadata" type="string">
      Optional metadata about the file from the user
    </ResponseField>
    <ResponseField name="identity_data" type="object">
      Identity data they decided to reveal about themselves
    </ResponseField>
  </Expandable>
</ResponseField>

## `verifyIdentity`

### Options

<ResponseField name="walletAddress" type="string" required>
  Valid Solana wallet address (Base58 format). Must have an existing user-key relation in the system.
</ResponseField>

<ResponseField name="disclosures" type="IdentityDisclosures" required>
  Object specifying which identity fields to request verification for. All 7 fields must be explicitly provided with boolean values. At least one field must be set to `true`.
  <Expandable>
    <ResponseField name="issuing_state" type="boolean">
      Country/state that issued the ID document.
    </ResponseField>
    <ResponseField name="name" type="boolean">
      Full name from ID document.
    </ResponseField>
    <ResponseField name="nationality" type="boolean">
      Nationality/citizenship.
    </ResponseField>
    <ResponseField name="date_of_birth" type="boolean">
      Date of birth.
    </ResponseField>
    <ResponseField name="passport_number" type="boolean">
      Document/passport number.
    </ResponseField>
    <ResponseField name="gender" type="boolean">
      Gender from ID document.
    </ResponseField>
    <ResponseField name="expiry_date" type="boolean">
      Document expiration date.
    </ResponseField>
  </Expandable>
</ResponseField>

### Response

#### New Session Created (201)

<ResponseField name="message" type="string">
  Confirmation that the identity verification session was created.
</ResponseField>

<ResponseField name="details" type="object">
  Session information and verification instructions.
  <Expandable>
    <ResponseField name="qrCodeUrl" type="string">
      URL for identity verification that can be opened in Self.xyz app.
    </ResponseField>
    <ResponseField name="walletAddress" type="string">
      The wallet address for this verification session.
    </ResponseField>
    <ResponseField name="selectedDisclosures" type="object">
      Object showing which identity fields were requested (only shows `true` values).
    </ResponseField>
    <ResponseField name="status" type="string">
      Current verification status: `"pending"`.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO timestamp when the session was created.
    </ResponseField>
    <ResponseField name="expiresAt" type="string">
      ISO timestamp when the session expires (24 hours from creation).
    </ResponseField>
    <ResponseField name="instructions" type="string">
      User-friendly instructions for completing verification.
    </ResponseField>
  </Expandable>
</ResponseField>

#### Existing Session Found (200)

<ResponseField name="message" type="string">
  Message indicating an existing pending session was found.
</ResponseField>

<ResponseField name="details" type="object">
  Existing session information.
  <Expandable>
    <ResponseField name="qrCodeUrl" type="string">
      URL for identity verification.
    </ResponseField>
    <ResponseField name="walletAddress" type="string">
      The wallet address for this verification session.
    </ResponseField>
    <ResponseField name="requestedDisclosures" type="object">
      All disclosure fields with boolean values.
    </ResponseField>
    <ResponseField name="status" type="string">
      Current verification status: `"pending"`.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO timestamp when the session was created.
    </ResponseField>
    <ResponseField name="expiresAt" type="string">
      ISO timestamp when the session expires.
    </ResponseField>
    <ResponseField name="instructions" type="string">
      Instructions for the existing session.
    </ResponseField>
  </Expandable>
</ResponseField>

## `checkIdentityStatus`

### Parameters

<ResponseField name="walletAddress" type="string" required>
  Valid Solana wallet address (Base58 format) to check verification status for.
</ResponseField>

### Response

<ResponseField name="message" type="string">
  Confirmation that the session was retrieved successfully.
</ResponseField>

<ResponseField name="details" type="object">
  Complete session information and verification data.
  <Expandable>
    <ResponseField name="walletAddress" type="string">
      The wallet address for this verification session.
    </ResponseField>
    <ResponseField name="status" type="'pending' | 'verified' | 'failed' | 'expired'">
      Current verification status.
    </ResponseField>
    <ResponseField name="createdAt" type="string">
      ISO timestamp when the session was created.
    </ResponseField>
    <ResponseField name="expiresAt" type="string">
      ISO timestamp when the session expires (24 hours from creation).
    </ResponseField>
    <ResponseField name="verifiedAt" type="string | null">
      ISO timestamp when verification was completed (null if not yet verified).
    </ResponseField>
    <ResponseField name="requestedDisclosures" type="object">
      Object showing which identity fields were requested for verification.
      <Expandable>
        <ResponseField name="issuing_state" type="boolean">
          Whether issuing state was requested.
        </ResponseField>
        <ResponseField name="name" type="boolean">
          Whether name was requested.
        </ResponseField>
        <ResponseField name="nationality" type="boolean">
          Whether nationality was requested.
        </ResponseField>
        <ResponseField name="date_of_birth" type="boolean">
          Whether date of birth was requested.
        </ResponseField>
        <ResponseField name="passport_number" type="boolean">
          Whether passport number was requested.
        </ResponseField>
        <ResponseField name="gender" type="boolean">
          Whether gender was requested.
        </ResponseField>
        <ResponseField name="expiry_date" type="boolean">
          Whether expiry date was requested.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="verifiedData" type="object | null">
      Contains the actual verified identity data (only present when status is 'verified').
      <Expandable>
        <ResponseField name="verified_issuing_state" type="string | null">
          Country/state that issued the ID document.
        </ResponseField>
        <ResponseField name="verified_name" type="string | null">
          Full name from the ID document.
        </ResponseField>
        <ResponseField name="verified_nationality" type="string | null">
          Nationality/citizenship.
        </ResponseField>
        <ResponseField name="verified_date_of_birth" type="string | null">
          Date of birth from the document.
        </ResponseField>
        <ResponseField name="verified_passport_number" type="string | null">
          Document/passport number.
        </ResponseField>
        <ResponseField name="verified_gender" type="string | null">
          Gender from the ID document.
        </ResponseField>
        <ResponseField name="verified_expiry_date" type="string | null">
          Document expiration date.
        </ResponseField>
        <ResponseField name="self_user_id" type="string">
          Unique identifier from Self.xyz verification system.
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

## `findWallet`

### Options

<ResponseField name="userID" type="string" optional>
  The user's ID.
</ResponseField>

<ResponseField name="walletAddress" type="string" optional>
  The user's wallet address.
</ResponseField>

### Response

<ResponseField name="success" type="boolean">
    Whether the request was successful.
</ResponseField>

<ResponseField name="message" type="string">
    A message describing the result of the request.
</ResponseField>

<ResponseField name="details" type="object">
    An object containing the details of the found wallet.
    <Expandable>
        <ResponseField name="searchType" type="string">
            The type of search that was performed.
        </ResponseField>
        <ResponseField name="searchValue" type="string">
            The value that was searched for.
        </ResponseField>
        <ResponseField name="relation" type="object">
            An object containing the details of the wallet relation.
            <Expandable>
                <ResponseField name="userId" type="string">
                    The user's ID.
                </ResponseField>
                <ResponseField name="walletAddress" type="string">
                    The user's wallet address.
                </ResponseField>
                <ResponseField name="createdAt" type="string">
                    The date the relation was created.
                </ResponseField>
                <ResponseField name="relationPDA" type="string">
                    The Program Derived Address for the relation.
                </ResponseField>
            </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>
