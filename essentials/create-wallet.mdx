## Quick Start

### Installation

```bash
npm install trust-engine-sdk
```

### Complete Example

<CodeGroup>
```typescript SDK (Recommended)
import { createWallet } from 'trust-engine-sdk';

async function createUserWallet() {
  try {
    const result = await createWallet({ 
      userID: 'user_12345' 
    });
    
    console.log('Wallet created:', result.details.walletAddress);
    console.log('Transaction:', result.details.transactionSignature);
    
    return result;
  } catch (error) {
    console.error('Failed to create wallet:', error.message);
    throw error;
  }
}

// Usage
createUserWallet()
  .then(result => console.log('Success:', result))
  .catch(error => console.error('Error:', error));
```

```bash cURL
curl -X POST http://localhost:3000/create-wallet \
  -H "Content-Type: application/json" \
  -d '{
    "userID": "user_12345"
  }'
```

```javascript Raw JavaScript
async function createWallet(userID) {
  try {
    const response = await fetch('http://localhost:3000/create-wallet', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ userID })
    });
    
    const data = await response.json();
    
    if (response.ok) {
      console.log('Wallet created:', data.details.walletAddress);
      return data;
    } else {
      throw new Error(data.message);
    }
  } catch (error) {
    console.error('Error creating wallet:', error.message);
    throw error;
  }
}

// Usage
createWallet('user_12345')
  .then(result => console.log('Success:', result))
  .catch(error => console.error('Failed:', error));
```
</CodeGroup>

## Parameters

<ResponseField name="userID" type="string" required>
  Unique identifier for the user
  
  **Constraints:** Maximum 50 characters
  
  Example: `"user_12345"`
</ResponseField>

---

## Response Examples

### Success Response (201)

<ResponseField name="message" type="string">
  Success confirmation message
  
  Example: `"Success! Your Crossmint wallet has been created and linked to your user ID."`
</ResponseField>

<ResponseField name="details" type="object">
  Wallet creation details
  
  <Expandable title="Details Object">
    <ResponseField name="walletAddress" type="string">
      The newly created Solana wallet address
      
      Example: `"9WzDXwBbmkg8ZTbNMqUxvQRAyrZzDsGYdLVL9zYtAWWM"`
    </ResponseField>
    
    <ResponseField name="chain" type="string">
      Blockchain network identifier
      
      Example: `"solana"`
    </ResponseField>
    
    <ResponseField name="transactionSignature" type="string">
      On-chain transaction signature for the wallet creation
      
      Example: `"2nBhEBYYvfaAe16UMNqRHre4YNSskvuYgx3M6E4JP1oDYvZEJHvoPzyUwkoB6VRYuVEMarHNr9RxJWnyb8FnFnBg"`
    </ResponseField>
    
    <ResponseField name="explorerUrl" type="string">
      Direct link to view the transaction on Solana Explorer
      
      Example: `"https://explorer.solana.com/tx/...?cluster=devnet"`
    </ResponseField>
  </Expandable>
</ResponseField>

## What This Endpoint Does

<Steps>
  <Step title="Creates Crossmint Wallet">
    Generates a new Solana wallet using Crossmint's API-key signing method
  </Step>
  <Step title="Stores User Relation">
    Creates an on-chain record linking your `userID` to the wallet address
  </Step>
  <Step title="Returns Details">
    Provides the wallet address and transaction confirmation
  </Step>
</Steps>
